137  // valori per l'encoding
138  uint_32 range = uint_32(pow(10.0, 9.0));
139  uint_32 low = 0;
140  uint_32 up = range;
141  byte b;
142 
143  // il bitwriter serve per scrivere in output le cifre da 0 a 9, quindi bastano 4 bit per cifra per codificarle
144  bitwriter bw(os);
145 
146  while (is.get(reinterpret_cast<char&>(b)))
147  encode_symbol(b, unsignedCouple, low, range, up, os, bw);
148 
149  // variabili per i calcoli successivi
150  uint_32 difference = 1;
151  float_64 tempLow;
152  float_64 tempTop;
153  float_64 cont = 0.0;
154 
155  // variabili d'appoggio, necessarie entrambe per il caso particolare che lo zero sia prima cifra di uno dei due
156  tempLow = low;
157  tempTop = up;
158 
159  // cerchiamo l'ordine di grandezza di low e top
160  while (tempLow > 10 || tempTop > 10){
161  tempLow /= 10;
162  tempTop /= 10;
163  cont++;
164  }
165 
166  // calcolo quante cifre ha il low e salvo in contLow per il for successivo
167  tempLow = 0.0;
168  uint_32 contLow = 0;
169  while (difference == 1){
170  tempLow = low / pow(10.0, cont);
171  difference = (uint_32)(up / (uint_32)pow(10, cont) - (uint_32)tempLow);
172  cont--;
173  contLow++;
174  }
175 
176  // calcolo valore finale da aggiungere al file
177  uint_32 final = (uint_32)(tempLow + difference / 2);
178  //cout << "finale:" << final << endl;
179 
180  // estrazione cifre finali e scrittura tramite bitwriter sul file
181  uint_32 lastDigit;
182  for (uint_32 i = 0; i < contLow; ++i){
183  lastDigit = (final *(uint_32)pow(10, i)) % (uint_32)pow(10, contLow) / (uint_32)pow(10, contLow - 1);
184  bw(lastDigit, 4);
185  }
